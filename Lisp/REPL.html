    end
  end

def run
  trap("SIGINT") { throw :exit }

  puts "ctrl-c to exit"

  catch(:exit) do
    loop do
      begin
        @input += io.readline prompt

        if open_count == close_count
          puts Lisp.eval input

          reset_input!
        end
      rescue Exception => e
        puts e.message

        reset_input!
      end
    end
  end
end
